\section{Contexte : Pourquoi l'énergie est critique ?}

\begin{frame}{Contexte : Pourquoi l'énergie est critique ?}

Les réseaux IoT sont fortement contraints par les ressources énergétiques.

\begin{block}{Problème}
Si certains appareils épuisent rapidement leur batterie, certaines tâches ne pourront plus être exécutées.
\end{block}

\begin{alertblock}{Objectif}
Concevoir un système intelligent capable de :
\begin{itemize}
    \item Réduire la consommation énergétique
    \item Garantir l'exécution des tâches
\end{itemize}
\end{alertblock}
\end{frame}

% Slide : Latence totale
\begin{frame}{Latence totale (Temps d'exécution et Congestion)}
\frametitle{Latence totale (Temps d'exécution et Congestion)}

Latence totale = \textbf{Temps de service pur} (calcul \& transmission) + \alert{\textbf{Délai d'attente (Queuing Delay)}}

\bigskip

\begin{columns}[T]  % T = aligner en haut
\begin{column}{0.50\textwidth}

\begin{block}{Temps de calcul perçu}
\[
T_{\text{calc}}^{r,l,j} = \sum_{p=1}^{P_{l-1}} A^j_{r,l,p} \cdot \frac{c^{l,p}_r}{e(j)} + \alert{W_{\text{comp}}^j}
\]
\end{block}

\begin{itemize}
\item $c^{l,p}_r$ \quad demande de calcul (cycles CPU / FLOPs)
\item $e(j)$ \quad efficacité de la ressource $j$
\item $\alert{W_{\text{comp}}^j}$ \, \textbf{Délai d'attente} si le processeur $j$ est surchargé par d'autres agents.
\end{itemize}

\end{column}

\begin{column}{0.48\textwidth}

\begin{block}{Temps de transmission perçu}
\[
T_{\text{trans}}^{r,l,i,j} = \sum_{p=1}^{P_{l-1}} A^i_{r,{l-1},p} \cdot A^j_{r,l,p} \cdot \frac{O^{r,l-1}_{i,j}}{\rho_i} \cdot \mathbf{1}_{i \neq j} + \alert{W_{\text{comm}}^j}
\]
\end{block}

\begin{itemize}
\item $O^{r,l-1}_{i,j}$ \quad volume de données à transmettre
\item $\rho_i$ \quad bande passante de l'appareil $i$
\item $\alert{W_{\text{comm}}^j}$ \, \textbf{Délai d'encombrement réseau} (bande passante saturée simultanément).
\end{itemize}

\end{column}
\end{columns}

\end{frame}


% Slide : Explication du Queuing Delay (Load Balancing)
\begin{frame}{Modélisation du réseau : Pourquoi intégrer les files d'attente ?}

\begin{columns}[T]
\begin{column}{0.49\textwidth}
\begin{block}{1. Environnement Idéal (Goulots ignorés)}
\begin{itemize}
    \item Capacité de traitement concurrente infinie implicite.
    \item Les agents exploitent le terminal le plus performant \textbf{sans subir de pénalité} de surcharge.
    \item \textbf{Résultat} : Déséquilibre extrême (un seul terminal de confiance puissant écope de tout le réseau).
\end{itemize}
\end{block}
\end{column}

\begin{column}{0.49\textwidth}
\begin{block}{2. Mode avec Queuing Delay (Réaliste)}
\begin{itemize}
    \item Traitement séquentiel (file FIFO) par appareil.
    \item \textbf{Congestion :} Si 5 tâches visent le même terminal, 4 patientent ($\uparrow W_{\text{comp}}$).
    \item \textbf{Load Balancing :} Les agents évitent les nœuds saturés pour minimiser la latence perçue et leur surconsommation ($E \propto T_{wait}$).
\end{itemize}
\end{block}
\end{column}
\end{columns}

\bigskip
\begin{alertblock}{Validation de notre approche MADDPG}
C'est avec la contrainte dynamique de congestion (Queuing Delay) que nous démontrons que nos agents apprennent d'eux-mêmes à \textbf{coopérer pour équilibrer la charge} (Load Balancing) sur tout le maillage IoT !
\end{alertblock}

\end{frame}

% Slide : Modèle d'énergie
\begin{frame}{Énergie dans la réalité : mesure et intégration}
\begin{block}{Formule générale}
    L'énergie consommée pendant une tâche est :
    \begin{equation*}
        E = P \cdot T
    \end{equation*}
    \begin{itemize}
        \item $P$ : puissance instantanée du dispositif, $T$ : durée totale d'exécution
        \item \textbf{Conclusion :} Plus le temps est long, plus l'énergie consommée est élevée.
    \end{itemize}
\end{block}

\centering
      \large\textbf{Énergie totale}
      \[
      E_{\text{total}} = E^{\text{calc}}_{\text{total}} + E^{\text{trans}}_{\text{total}} = P_i^{\text{comp}} \times T_{\text{calc}} + P_i^{\text{tx}} \times T_{\text{trans}} 
      \]

\begin{columns}[T]
    \begin{column}{0.50\textwidth}
        \centering
        \textbf{Énergie de calcul}
        \[
        E^{\text{calc}}_{\text{total}} = \sum_{i,r,l} A^i_{r,l} \, P_i^{\text{comp}} \, \frac{c^l_r}{e(i)}
        \]

        \centering
        \textbf{Énergie de transmission}
        \[
        E^{\text{trans}}_{\text{total}} = \sum_{i,j,r,l} A^i_{r,l} A^j_{r,l+1} P_i^{\text{tx}} \frac{O^l_r}{\rho_i} \mathbf{1}_{i\neq j}
        \]
    \end{column}
    
    \begin{column}{0.50\textwidth}
        \centering
        \textbf{Énergie de transmission}
        \[
        E^{\text{trans}}_{\text{total}} = \sum_{i,j,r,l} A^i_{r,l} A^j_{r,l+1} P_i^{\text{tx}} \frac{O^l_r}{\rho_i} \mathbf{1}_{i\neq j}
        \]
    \end{column}
\end{columns}

\end{frame}

\section{Deux rôles différents de l’énergie}

\begin{frame}{Cœur conceptuel de notre approche}
\centering
\Huge L'énergie joue deux rôles bien distincts

\bigskip

\begin{columns}[T]
\begin{column}{0.48\textwidth}
    \textbf{Objectif d’optimisation} 
    \vspace{0.2cm}
    \\ (le ``quoi'')
    \vspace{0.2cm}
    \normalsize
    \begin{itemize}
        \item ce que l’IA essaie d’optimiser
        \item Rôle de l’énergie : Économie
    \end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
    \textbf{Contrainte physique} 
    \vspace{0.2cm}
    \\ (le ``comment'')
    \vspace{0.2cm}
    \normalsize
    \begin{itemize}
        \item ce qui limite / interdit certaines actions
        \item Rôle de l’énergie : Sécurité
    \end{itemize}
\end{column}
\end{columns}
\end{frame}

% -------------------------------------------------------------------------
\begin{frame}{1. Énergie dans la fonction de récompense}
\frametitle{1. Énergie comme terme dans le reward}

\textbf{Objectif} : contraindre les agents à trouver des stratégies qui soient \textbf{à la fois rapides} et \textbf{très économes en énergie}, afin de maximiser la durée de vie des batteries des objets IoT.

Reward typique :
\[ r = -( \text{latence} + \text{énergie consommée}) \]

\begin{itemize}
    \item Pousse l’agent à choisir l’option \alert{la plus économe} parmi les solutions valides
\end{itemize}

\bigskip
\centering
\textbf{Message à l’agent} : « Même si la batterie est quasi-pleine, encouragé à choisir l'option la plus économe. »
\end{frame}

% -------------------------------------------------------------------------
\begin{frame}{2. Énergie comme contrainte dure (Hard Constraint)}
\frametitle{2. Énergie comme contrainte dure}

\begin{block}{Règle de survie}
Avant chaque action :

\[ \text{batterie\_restante} < \text{coût\_énergie\_tâche} \quad \Longrightarrow \quad \text{échec immédiat} \]
\end{block}

\bigskip
Conséquences :
\begin{itemize}
    \item Pénalité très forte si violation : \alert{$-500$}
    \item L’agent apprend très vite que certains nœuds deviennent \alert{interdits}
\end{itemize}
\end{frame}

% -------------------------------------------------------------------------
\begin{frame}{En résumé – la double casquette}
\centering
\Large
\begin{tabular}{c}
\alert{Contrainte} $\quad\Longrightarrow\quad$ \textbf{empêche de dépasser le budget } \\[0.8em]
\alert{Objectif}   $\quad\Longrightarrow\quad$ \textbf{consommer le strict minimum}
\end{tabular}

\\[1.2em]
\large$\Downarrow$ \\[0.6em]
\centering
{Cette combinaison qui permet à notre système de garantir que les tâches seront terminées (Contrainte) tout en étant le plus performant possible (Objectif).}


\bigskip
\Large Garantir la \alert{faisabilité} \quad ET \quad maximiser la \alert{performance globale}
\end{frame}

\begin{frame}{2. Objectifs et formules des deux réseaux en DDPG / MADDPG}
\begin{columns}[T]
% ------------------- COLONNE GAUCHE : ACTOR -------------------
\column{0.48\textwidth}
\begin{block}{Objectif de l'Actor (Policy Network)}
    \textbf{Maximiser} la Q-value estimée par le Critic\\
    $\Rightarrow$ Trouver la meilleure action possible dans chaque état,  
\end{block}

\smallskip
L'Actor apprend la \textbf{politique} déterministe optimale $\pi(s)$.

\begin{equation*}
    \mathcal{L}_{\text{actor}} = - \mathbb{E} \left[ Q(s, \pi(s)) \right]
\end{equation*}
\vspace{0.3cm}
\begin{itemize}
    \item Si $Q(s, a)$ s'améliore (moins négatif) → loss Actor \textbf{se stabilise}
\end{itemize}

% ------------------- COLONNE DROITE : CRITIC -------------------
\column{0.48\textwidth}
\begin{block}{Objectif du Critic (Q-Network)}
    \textbf{Minimiser} l'erreur de prédiction de la Q-value\\
\end{block}

\smallskip
Le Critic \textbf{évalue} la politique actuelle à l'aide de la fonction de valeur $Q$.

\begin{equation*}
    \mathcal{L}_{\text{critic}} = \mathbb{E} \Bigl[ \bigl( y - Q(s, a) \bigr)^2 \Bigr]
\end{equation*}
\begin{itemize}
    \item $y$ = cible (Bellman) : reward immédiat + valeur future estimée
\end{itemize}
\end{columns}
\end{frame}
