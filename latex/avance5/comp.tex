\begin{frame}{Compromis Énergie-Latence — Interprétation}

\begin{alertblock}{Résultat clé}
\centering
\textbf{Trade-off fondamental imposé par la physique}
\end{alertblock}

\vspace{0.3cm}

\begin{block}{Interprétation}
\begin{itemize}
    \item Réduction fréquence CPU → tâches plus lentes
    \item Accumulation dans les files → hausse latence
    \item Choix optimal appris :
    \begin{itemize}
        \item Survivre (énergie) > rapidité (latence)
    \end{itemize}
    \item Compromis inévitable dans systèmes réels
\end{itemize}
\end{block}

\end{frame}

\begin{itemize}
    \item Modèle DVFS ($P \propto f^3$) impose des contraintes sévères
    \item Exécution à pleine fréquence → échec (batterie vide)
    \item Les agents apprennent à \textbf{adapter leur vitesse}
    \item Passage de 0\% à $\sim$100\% de succès
\end{itemize}

\begin{frame}{Conclusion Générale}

\begin{alertblock}{Contribution principale}
\centering
\textbf{Intégration d’un modèle énergétique réaliste change radicalement le comportement appris}
\end{alertblock}

\vspace{0.3cm}

\begin{block}{Insights clés}
\begin{itemize}
    \item Baseline : optimisation facile mais irréaliste
    \item DVFS : environnement contraint → apprentissage complexe
    \item Découverte d’un \textbf{trade-off énergie-latence}
    \item Les agents apprennent à \textbf{survivre avant d’optimiser}
    \item Résultat : système réaliste mais avec compromis
\end{itemize}
\end{block}
