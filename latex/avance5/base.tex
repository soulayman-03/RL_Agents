\begin{frame}{Compromis Énergie-Latence — Interprétation}

\begin{alertblock}{Résultat clé}
\centering
\textbf{Sacrifice de la latence pour maximiser l’économie d’énergie}
\end{alertblock}

\vspace{0.3cm}

\begin{block}{Interprétation}
\begin{itemize}
    \item Trade-off fondamental appris par les agents
    \item Réduction de fréquence CPU → exécution plus lente et Moins de consommation
    \item Optimisation guidée par la fonction de récompense globale
    \item Choix optimal : \textbf{efficacité énergétique > rapidité}
\end{itemize}
\end{block}

\end{frame}
