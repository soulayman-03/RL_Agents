\section{Contexte et état de l'art}

%---------------------------------------------------
\begin{frame}{Problème étudié}
\begin{itemize}
    \item Allocation distribuée de couches de réseaux de neurones convolutifs (CNN)
    \item Environnement IoT fortement contraint (ressources limitées)
    \item Approche multi-agent coopérative
    \item Objectif principal : \textbf{minimiser la latence d'inférence}
\end{itemize}

\centering
{\large \textcolor{red}{Edge Intelligence} $\quad\Rightarrow\quad$ \textcolor{blue}{Partitionnement intelligent du modèle}}
\end{frame}

        \item Forte variance entre agents et au sein d’un même agent
        \item Oscillations violentes même en fin d'entraînement → signe de \textbf{divergence ou instabilité majeure}
        \item Moving average (50) ne se stabilise jamais → apprentissage complètement échoué
    \end{itemize}

%---------------------------------------------------
\begin{frame}{Synthèse – Échec de l'expérience (HugeCNN / contraintes strictes ?)}
\begin{alertblock}{Résumé de l'échec}
\begin{itemize}
    \item \textbf{Taux de succès = 0.0\%} sur 5000 épisodes d'évaluation
    \item Tous les épisodes (5000/5000) ont échoué
    \item Team Return reste extrêmement négatif et instable (-40 à -100)
    \item Reward par agent très faible et hautement variable → pas de coordination apprise
    \item TD Loss reste élevée (~5000–7000) → critiques n'ont pas convergé
\end{itemize}
\end{alertblock}

\begin{block}{Hypothèse probable sur la cause}
\begin{itemize}
    \item Modèle HugeCNN beaucoup plus lourd → contraintes ressources (mémoire, calcul, énergie) trop strictes
\end{itemize}
\end{block}

\small\centering
$\Rightarrow$ Cette expérience montre les limites actuelles quand on augmente la complexité (modèle + contraintes)
\end{frame}

            \item Agent 0 : CNN-15 (léger)
            \item Agent 1 : MiniResNet (intermédiaire)
            \item Agent 2 : HugeCNN (très lourd)
        \end{itemize}

        \begin{itemize}

            \item Temps d'entraînement : 5 h 59 min
        \end{itemize}
