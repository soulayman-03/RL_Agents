\section{Limites identifiées}

%---------------------------------------------------
\begin{frame}{Limites de l'approche actuelle}
\begin{block}{Limites principales}
\begin{itemize}
    \item Optimisation mono-objectif : \textbf{uniquement la latence}
    \item \textbf{Aucune prise en compte de l'énergie} (critique en IoT)
    \item Confidentialité traitée de façon \textbf{binaire} : 
      \begin{itemize}
          \item Exposition complète interdite ou autorisée
          \item Absence de différenciation des niveaux de confiance / fiabilité des devices (pas de trust-aware constraint)
      \end{itemize}
    \item Absence de comparaison avec des algorithmes multi-agents plus récents
\end{itemize}
\end{block}

 \bigskip
$\Rightarrow$ Motivation forte pour des extensions réalistes et performantes
\end{frame}

\section{Extension 1 – Intégration de la contrainte énergétique}

%---------------------------------------------------
\begin{frame}{Motivation – Prise en compte de l'énergie}
En environnement IoT réel :

\begin{itemize}
    \item Dispositifs majoritairement sur batterie
    \item Énergie = ressource la plus critique (souvent plus que la latence)
    \item Optimiser uniquement la latence $\Rightarrow$ décharges prématurées
\end{itemize}

$\Rightarrow$ Nécessité d'intégrer la durabilité énergétique
\end{frame}

%---------------------------------------------------
\begin{frame}{Deux approches expérimentales}
\begin{block}{Expérience A – Contrainte stricte (hard constraint)}
\[
\min \text{Latence} \quad \text{s.t.} \quad \sum_{r^*,l} A^i_{r^*,l} E^l_{r^*} \le \bar{E}_i \quad \forall i
\]
\end{block}

\begin{columns}[T]
    \column{0.55\textwidth}
        \begin{block}{Signification pratique}
        \begin{itemize}
            \item L’objectif principal reste la \textbf{minimisation de la latence}
            \item Mais \textbf{aucun dépassement énergétique n’est toléré}
            \item Résultat : l’agent ne peut choisir que des assignments \textbf{physiquement possibles}
        \end{itemize}
        \end{block}
    
        \column{0.45\textwidth}
        \begin{alertblock}{Exemple concret}
            Supposons : \\
            $\bar{E}_i = 80$ unités restantes \\
            Couche à placer : coût estimé = 95 unités \\
            $\Rightarrow$ 80 < 95 $\Rightarrow$ device $i$ \textbf{invalide} \\
            Si tous les devices deviennent invalides $\Rightarrow$ \textbf{échec de l’épisode}
        \end{alertblock}
    \end{columns}
\end{frame}

\begin{frame}{Modélisation énergétique détaillée – Calcul et Transmission}
\frametitle{Modélisation réaliste de la consommation énergétique}

\begin{block}{A) Énergie de calcul (partie computation)}
\[
E^{calc}_{r^*,l,i} = P_i^{comp} \times T^{calc}_{r^*,l,i}
\]
où :
\begin{itemize}
    \item $P_i^{comp}$ : puissance CPU/GPU du device $i$ (en Watts)
    \item $T^{calc}_{r^*,l,i}$ : temps d'exécution de la couche $l$ du modèle $r^*$ sur le device $i$
\end{itemize}

\small
Temps de calcul approximé par :
\[
T^{calc}_{r^*,l,i} = \frac{c^l_{r^*}}{e(i)}
\]
avec $c^l_{r^*}$ = complexité de la couche (FLOPs ou MACs), $e(i)$ = efficacité du device $i$ (FLOPs/s)

Consommation totale calcul :
\[
E^{calc}_{total} = \sum_{i,r^*,l} A^i_{r^*,l} \, P_i^{comp} \, \frac{c^l_{r^*}}{e(i)}
\]
\end{block}
\end{frame}

\begin{frame}{Modélisation énergétique détaillée – Calcul et Transmission}
\frametitle{Modélisation réaliste de la consommation énergétique}

\begin{block}{B) Énergie de transmission (partie communication)}
\[
E^{trans}_{r^*,l,i} = P_i^{tx} \times T^{trans}_{r^*,l}
\]
où :
\begin{itemize}
    \item $P_i^{tx}$ : puissance de transmission du device $i$ (en Watts)
    \item $T^{trans}_{r^*,l}$ : temps de transmission des activations de la couche $l$
\end{itemize}

Temps de transmission :
\[
T^{trans}_{r^*,l} = \frac{O^l_{r^*}}{\rho_i}
\]
avec $O^l_{r^*}$ = taille des données sortantes, $\rho_i$ = débit uplink du device $i$

Consommation totale transmission :
\[
E^{trans}_{total} = \sum_{r^*,l,i,j} A^i_{r^*,l} \, A^j_{r^*,l+1} \, P_i^{tx} \, \frac{O^l_{r^*}}{\rho_i} \, \mathbf{1}_{i \neq j}
\]
\end{block}
\end{frame}
%---------------------------------------------------
\begin{frame}{Synthèse – Pourquoi cette modélisation énergétique ?}
\frametitle{Avantages et limites de la modélisation}

\begin{columns}[T]
    \column{0.5\textwidth}
    \begin{block}{\textcolor{teal}{Avantages}}
    \begin{itemize}
        \item Réalisme physique : sépare calcul et communication (très important en IoT)
        \item Paramétrable : $\alpha_{comp}$ et $\alpha_{comm}$ permettent d’ajuster l’importance relative
        \item Compatible avec la littérature edge computing (modèles classiques)
        \item Facile à intégrer dans l’environnement (coût calculé avant chaque action)
    \end{itemize}
    \end{block}

    \column{0.5\textwidth}
    \begin{alertblock}{\textcolor{red}{Limites et simplifications}}
    \begin{itemize}
        \item Linéaire → ne capture pas les effets non-linéaires (DVFS, leakage, etc.)
        \item Puissance constante ($P_i^{comp}$, $P_i^{tx}$) → pas de variation dynamique
        \item Débit $\rho_i$ fixe → pas de congestion réseau
        \item Pas de prise en compte du idle power ou du réveil
    \end{itemize}
    \end{alertblock}
\end{columns}

\small
\begin{block}{Conclusion}
Cette modélisation est un bon compromis entre réalisme et complexité computationnelle.  
Elle permet de montrer que l’énergie de transmission est souvent critique → justifie l’utilisation de $\alpha_{comm}$ élevé dans les expériences.
\end{block}
\end{frame}

\section{Extension 2 – Contrainte de confiance pondérée}
%---------------------------------------------------
\begin{frame}{Extension 2 – Contrainte de confiance pondérée (Trust Score)}
\frametitle{Contrainte de confiance pondérée – Trust-aware allocation}

\begin{block}{Problème actuel dans le modèle de base}
\begin{itemize}
    \item La confidentialité est traitée de façon \textbf{binaire} :
    \begin{itemize}
        \item Exposition complète d’une couche → interdite
        \item Pas d’exposition → autorisée
    \end{itemize}
    \item Tous les devices sont considérés comme \textbf{équivalents} en termes de fiabilité
\end{itemize}
\end{block}
\bigskip

\begin{alertblock}{Réalité des systèmes IoT distribués}
\begin{itemize}
    \item Certains devices sont plus fiables (privés, sécurisés, owned)
    \item D’autres sont moins fiables (publics, partagés, potentiellement compromis)
    \item Attaques internes possibles même dans le réseau
\end{itemize}
\end{alertblock}
\end{frame}

%---------------------------------------------------
\begin{frame}{Proposition – Intégration d’un score de confiance}
\frametitle{Contrainte de confiance pondérée}
\begin{block}{1) Pondérée par violation de niveau}
Idée : comparaison directe des niveaux de confidentialité
\begin{itemize}
    \item Layer possède un \textbf{privacy\_level} (ex. 0 = public, 1 = privé, ou plusieurs niveaux)
    \item Device possède un \textbf{privacy\_clearance} (niveau d’autorisation max)
\end{itemize}
$\Rightarrow$ Plus un layer privé est mis sur un device peu autorisé, plus la pénalité est forte.
\end{block}
\end{frame}

\begin{frame}{Contrainte de confiance pondérée}
    \begin{block}{2) Méthode actuelle – Seuil minimal de confiance (trust)}
    Idée : on utilise un score de confiance continu du device au lieu d’un niveau fixe
    \begin{itemize}
        \item Device possède un \textbf{trust\_score} $\in [0, 1]$ (Historique de comportement : incidents, anomalies, tentatives d’intrusion, taux d’échecs.)
        \item On convertit le privacy\_level du layer en \textbf{trust requis} :
        \[
        \text{trust\_required} = \text{trust\_min\_for\_max\_privacy} \times \frac{\text{privacy\_level}}{\text{privacy\_max\_level}}
        \]
        \item \text{Trust\_score} \ge \text{trust\_requis : pour exécuter certaines couches sensibles} \\
        $\Rightarrow$ un device doit avoir un score de confiance suffisant pour être candidat
    \end{itemize}
    \end{block}
\end{frame}
%---------------------------------------------------
\begin{frame}{Synthèse – Évolution de la confidentialité}
\frametitle{Évolution de la confidentialité dans le modèle}

\begin{table}[!h]
\footnotesize
\centering
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{lcc}
\toprule
Niveau & Modèle de base (contrainte binaire) & Avec trust pondéré \\
\midrule
Confidentialité & Binaire (exposition oui/non) & \textbf{Graduelle et adaptative} \\
Fiabilité des devices & Équivalente & \textbf{Différenciée (Trust$_i$)} \\
Attaques internes & Faible protection & \textbf{Meilleure défense} \\
Réalisme IoT & Limité & \textbf{Élevé} \\
\bottomrule
\end{tabular}
\caption{Progression vers un modèle plus sécurisé et réaliste}
\end{table}

\small
\begin{block}{Conclusion}
Ajouter un score de confiance pondéré transforme la confidentialité d’une règle binaire en une propriété \textbf{dynamique et contextuelle} — compatible zero-trust et très pertinente pour publication.
\end{block}
\end{frame}

\section{Extension 3 – Remplacement de MADDPG par MATD3}

%---------------------------------------------------
\begin{frame}{Pourquoi passer à MATD3 ?}
MADDPG souffre de problèmes connus en environnement continu :

\begin{itemize}
    \item Surestimation systématique des valeurs Q (overestimation bias)
    \item Instabilité lors de l'apprentissage
\end{itemize}

 \bigskip
MATD3 (Multi-Agent Twin Delayed DDPG) apporte :

\begin{itemize}
    \item Double critic $\quad\Rightarrow\quad \min(Q_1, Q_2)$
    \item Delayed policy updates
    \item Target policy smoothing (bruit ajouté à la cible)
\end{itemize}

$\Rightarrow$ Réduction forte du biais de surestimation + meilleure stabilité
\end{frame}

%---------------------------------------------------
\begin{frame}{Comparaison MADDPG vs MATD3}
\begin{table}[]
\footnotesize
\centering
\begin{tabular}{lcc}
\toprule
Critère                  & MADDPG              & MATD3                  \\
\midrule
Nombre de critics        & 1                   & 2 (min)                \\
Mise à jour de la politique & simultanée          & retardée (delayed)     \\
Sensibilité au bruit     & plus élevée         & réduite (smoothing)    \\
Stabilité globale        & moyenne             & meilleure              \\
\bottomrule
\end{tabular}
\end{table}

\centering
\textbf{MATD3 est aujourd'hui considéré comme l'un des SOTA en multi-agent continu}\\

\\
On attend de MATD3 :

\begin{itemize}
    \item Réduction de la variance des courbes d'apprentissage
    \item Meilleure convergence (plus rapide et plus stable)
    \item Taux de succès final potentiellement supérieur
    \item Meilleure robustesse face aux contraintes supplémentaires
\end{itemize}

\end{frame}

\section{Perspective}
\begin{frame}{Perspective : Limites du modèle actuel}

\textbf{Objectif actuel :}
\[
\min T_{total}
\]

\textbf{Implications :}
\begin{itemize}
    \item L’optimisation se concentre uniquement sur la rapidité d’exécution.
    \item La consommation énergétique n’est pas prise en compte explicitement.
    \item Les dispositifs rapides mais très énergivores peuvent être surutilisés.
    \item Risque d’épuisement de la batterie et de congestion du réseau.
\end{itemize}

\textbf{Conclusion :}
\begin{block}{}
Optimisation basée uniquement sur la performance, sans considération de durabilité.
\end{block}

\end{frame}

\begin{frame}{Perspective : Optimisation éco-énergétique}

\textbf{Nouvelle formulation multi-objectif proposée :}

\[
\min \left( T_{total} + E_{total} \right)
\]

\textbf{Interprétation :}
\begin{itemize}
    \item $T_{total}$ : latence totale (calcul + communication)
    \item $E_{total}$ : énergie totale (calcul + transmission)
\end{itemize}

\textbf{Transformation :}

\begin{block}{}
Passage d’une optimisation purement basée sur la performance  
à une optimisation intégrant performance et durabilité.
\end{block}
\end{frame}
